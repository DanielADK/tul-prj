@startuml

cloud "Backend Systém" {
    [Backend Server]
}

node "Čipový Server" {
    database "Databáze Čipového Serveru"
    [API Čipového Serveru]
    node "Zařízení pro Příchody" as Prichody
    node "Zařízení pro Odchody" as Odchody
}

[Backend Server] <-left- [API Čipového Serveru] : Získává\ndata
[API Čipového Serveru] <-down- "Databáze Čipového Serveru" : Získává\ndata
Prichody -up-> "Databáze Čipového Serveru" : Ukládá\npříchody
Odchody -up-> "Databáze Čipového Serveru" : Ukládá\nodchody

note right of Prichody
  Zařízení pro evidenci příchodů
  Umístěno před dveřmi do školy
end note

note right of Odchody
  Zařízení pro evidenci odchodů
  Umístěno před dveřmi ze školy
end note

@enduml
